# lumina-ui

> Fascinating ui components made for react

[![NPM](https://img.shields.io/npm/v/lumina-ui.svg)](https://www.npmjs.com/package/lumina-ui) [![JavaScript Style Guide](https://img.shields.io/badge/code_style-standard-brightgreen.svg)](https://standardjs.com)

## Install

```bash
npm install --save lumina-ui
```

## Usage

```tsx
import {
  AutoCompleteSelect,
  Button,
  SubmitDialogBox,
  Checkbox,
  CustomChipWithDelete,
  CustomGroupButton,
  CustomLoader,
  CustomPagination,
  CustomSnackbar,
  CustomTabComponent,
  DatePicker,
  DeleteDialogBox,
  Dropdown,
  Input,
  Radio,
  Select,
  SideNav,
  TextArea,
  ToggleSwitch,
  Divider,
  Avatar,
  Stepper,
  CheckBoxGroup,
  MenuBar
} from "lumina-ui";
import "lumina-ui/dist/index.css";

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
